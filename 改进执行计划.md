## 二、第一阶段：内存安全（P0-紧急）

### 2.1 问题分析

**风险点**：
1. 跨线程信号槽传递裸指针，接收者需负责delete，易造成内存泄漏
2. 线程异常终止时动态分配的缓冲区无法释放
3. 未使用智能指针管理动态对象

**关键文件**：
- `xcp/xcp_thread.cpp` - RESDataReady信号传递
- `xcp/xcp_polling_thread.cpp` - pollDataForRecord信号
- `xcp/charpamcheck.cpp` - data_last/data_new缓冲区
- `xcp/xcpmaster.cpp` - DAQ数据处理

### 2.2 执行步骤

**文件**：`XCP_COMM/common/smart_ptr.h`（新建）

```cpp
#ifndef SMART_PTR_H
#define SMART_PTR_H

#include <QSharedPointer>
#include <memory>

// 线程安全的字节数组智能指针
using ByteArrayPtr = QSharedPointer<quint8>;

inline ByteArrayPtr makeByteArray(int size) {
    return ByteArrayPtr(new quint8[size], [](quint8* p) { delete[] p; });
}

// 线程安全的字符数组智能指针
using CharArrayPtr = QSharedPointer<char>;

inline CharArrayPtr makeCharArray(int size) {
    return CharArrayPtr(new char[size], [](char* p) { delete[] p; });
}

#endif // SMART_PTR_H
```

```cpp
#include "param_table_model.h"
#include <QTimer>

ParamTableModel::ParamTableModel(QObject *parent)
    : QAbstractTableModel(parent)
{
}

int ParamTableModel::rowCount(const QModelIndex &parent) const
{
    return parent.isValid() ? 0 : m_params.count();
}

int ParamTableModel::columnCount(const QModelIndex &parent) const
{
    return parent.isValid() ? 0 : ColCount;
}

QVariant ParamTableModel::data(const QModelIndex &index, int role) const
{
    if (!index.isValid() || index.row() >= m_params.count())
        return QVariant();
    
    if (role == Qt::DisplayRole) {
        PARAM *pam = m_params.at(index.row());
        switch (index.column()) {
        case ColName:
            return pam->Name;
        case ColValue:
            return QString::number(m_cachedValues[index.row()], 'f', 3);
        case ColUnit:
            return pam->Unit;
        }
    }
    
    return QVariant();
}

QVariant ParamTableModel::headerData(int section, Qt::Orientation orientation, int role) const
{
    if (orientation == Qt::Horizontal && role == Qt::DisplayRole) {
        switch (section) {
        case ColName: return tr("参数");
        case ColValue: return tr("值");
        case ColUnit: return tr("单位");
        }
    }
    return QVariant();
}

void ParamTableModel::setParams(const QList<PARAM*> &params)
{
    beginResetModel();
    m_params = params;
    m_cachedValues.resize(params.count());
    endResetModel();
}

void ParamTableModel::updateValue(int row, double value)
{
    if (row >= 0 && row < m_cachedValues.size()) {
        m_cachedValues[row] = value;
        emit dataChanged(index(row, ColValue), index(row, ColValue));
    }
}

void ParamTableModel::updateAllValues()
{
    for (int i = 0; i < m_params.count(); ++i) {
        m_cachedValues[i] = m_params.at(i)->getValue();
    }
    emit dataChanged(index(0, ColValue), 
                     index(m_params.count() - 1, ColValue));
}
```

```cpp
// 添加异步写入队列
#include <QQueue>
#include <QMutex>
#include <QWaitCondition>

class AsyncLogWriter : public QThread
{
public:
    void enqueue(const LogMessage &msg) {
        QMutexLocker locker(&m_mutex);
        m_queue.enqueue(msg);
        m_condition.wakeOne();
    }
    
protected:
    void run() override {
        QFile file(m_filePath);
        file.open(QIODevice::WriteOnly | QIODevice::Append);
        
        while (!isInterruptionRequested()) {
            LogMessage msg;
            {
                QMutexLocker locker(&m_mutex);
                if (m_queue.isEmpty()) {
                    m_condition.wait(&m_mutex, 100);
                    continue;
                }
                msg = m_queue.dequeue();
            }
            
            // 写入文件
            file.write(msg.toBytes());
        }
        
        file.close();
    }
    
private:
    QQueue<LogMessage> m_queue;
    QMutex m_mutex;
    QWaitCondition m_condition;
    QString m_filePath;
};
```
