```cpp
#include <memory>

class Node {
    std::unique_ptr<char[]> name;
    Node* parentNode;  // 非拥有指针，保持原始指针
    std::vector<std::unique_ptr<Node>> childNodes;
    std::vector<std::unique_ptr<Item>> optItems;
    
public:

        child->parentNode = this;
        Node* ptr = child.get();
        childNodes.push_back(std::move(child));
        return ptr;
    }
    
    // 返回原始指针供外部使用
    Node* child(int row) const {
        if (row >= 0 && row < childNodes.size()) {
            return childNodes[row].get();
        }
        return nullptr;
    }
```

```cpp
        CompuMethod, RecordLayout, AxisPts, Function, Group

    std::vector<AstNode*> findChildrenByType(Type type);
```

#### 4.2.3 访问者模式

```cpp
// core/visitor/data_extractor.h
#pragma once
#include "core/ast/ast_node.h"
#include <functional>

namespace a2l {

class DataExtractor {
public:
    using Callback = std::function<void(const AstNode&)>;
    
    void setCharacteristicCallback(Callback cb) { m_charCallback = std::move(cb); }
    void setMeasurementCallback(Callback cb) { m_measCallback = std::move(cb); }
    
    void visit(const AstNode& node) {
        switch (node.type) {
            case AstNode::Type::Characteristic:
                if (m_charCallback) m_charCallback(node);
                break;
            case AstNode::Type::Measurement:
```

### 5.1 内存池优化

```cpp
// core/memory/ast_node_pool.h
#pragma once
#include <memory>
#include <stack>

namespace a2l {

template<typename T>
class ObjectPool {
public:
    template<typename... Args>
    std::unique_ptr<T, std::function<void(T*)>> acquire(Args&&... args) {
        if (m_pool.empty()) {
            return std::unique_ptr<T, std::function<void(T*)>>(
                new T(std::forward<Args>(args)...),
                [this](T* ptr) { release(ptr); }
            );
        }
        auto obj = m_pool.top();
        m_pool.pop();
        return std::unique_ptr<T, std::function<void(T*)>>(
            obj, [this](T* ptr) { release(ptr); }
        );
    }
    
    void preAllocate(size_t count) {
        for (size_t i = 0; i < count; ++i) {
            m_pool.push(new T());
        }
    }

private:
    void release(T* obj) {
        obj->reset();  // 假设T有reset方法
        m_pool.push(obj);
    }
    
    std::stack<T*> m_pool;
};

} // namespace a2l
```

```powershell
cd D:\lyx\code\TOxMaster\XCP_COMM
```
